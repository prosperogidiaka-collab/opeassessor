
import React, { useState, useEffect, useCallback, useRef } from 'react';
import { Quiz, Submission } from '../types';
import { Card, Button } from './Layout';
import { storageService } from '../services/storageService';

interface Props {
  quiz: Quiz;
  studentName: string;
  studentEmail: string;
  onFinish: (submission: Submission) => void;
}

export const QuizInterface: React.FC<Props> = ({ quiz, studentName, studentEmail, onFinish }) => {
  const [currentIdx, setCurrentIdx] = useState(0);
  const [answers, setAnswers] = useState<Record<string, string>>({});
  const [timeLeft, setTimeLeft] = useState(quiz.durationMinutes * 60);
  const [isFinished, setIsFinished] = useState(false);
  const [showConfirmation, setShowConfirmation] = useState(false);
  const [submissionData, setSubmissionData] = useState<Submission | null>(null);
  
  // Anti-cheating states
  const [warningTimer, setWarningTimer] = useState<number | null>(null);
  
  const startTime = useRef(Date.now());
  const answersRef = useRef(answers);
  const autoSubmitTimeout = useRef<any | null>(null);

  useEffect(() => {
    answersRef.current = answers;
  }, [answers]);

  const handleSubmit = useCallback((auto = false) => {
    if (isFinished) return;
    setIsFinished(true);

    const correctCount = quiz.questions.reduce((acc, q) => {
      return acc + (answersRef.current[q.id] === q.answer ? 1 : 0);
    }, 0);

    const finalGrade = (correctCount / quiz.questions.length) * quiz.maxScore;

    const submission: Submission = {
      id: Math.random().toString(36).substr(2, 9),
      quizId: quiz.id,
      studentName,
      studentEmail,
      answers: answersRef.current,
      score: correctCount,
      totalPossible: quiz.questions.length,
      finalGrade,
      startTime: startTime.current,
      endTime: Date.now(),
      autoSubmitted: auto
    };

    storageService.saveSubmission(submission);
    setSubmissionData(submission);
    setShowConfirmation(true);
  }, [quiz, studentName, studentEmail, isFinished]);

  // Tab visibility detection
  useEffect(() => {
    const handleVisibilityChange = () => {
      if (document.hidden) {
        // Start 5 second countdown
        setWarningTimer(5);
        autoSubmitTimeout.current = setTimeout(() => {
          handleSubmit(true);
        }, 5000);
      } else {
        // Return detected - clear timeout and warning
        if (autoSubmitTimeout.current) {
          clearTimeout(autoSubmitTimeout.current);
          autoSubmitTimeout.current = null;
        }
        setWarningTimer(null);
      }
    };

    document.addEventListener('visibilitychange', handleVisibilityChange);
    return () => document.removeEventListener('visibilitychange', handleVisibilityChange);
  }, [handleSubmit]);

  // Warning countdown interval
  useEffect(() => {
    let interval: any;
    if (warningTimer !== null && warningTimer > 0) {
      interval = setInterval(() => {
        setWarningTimer(prev => (prev && prev > 0 ? prev - 1 : 0));
      }, 1000);
    }
    return () => clearInterval(interval);
  }, [warningTimer]);

  // Primary Exam Timer
  useEffect(() => {
    if (showConfirmation || isFinished) return;
    if (timeLeft <= 0) {
      handleSubmit();
      return;
    }
    const timer = setInterval(() => setTimeLeft(prev => prev - 1), 1000);
    return () => clearInterval(timer);
  }, [timeLeft, handleSubmit, showConfirmation, isFinished]);

  // Warning Overlay UI
  if (warningTimer !== null && !isFinished) {
    return (
      <div className="fixed inset-0 z-[100] flex items-center justify-center bg-rose-600 p-6 text-white text-center">
        <div className="max-w-md animate-in fade-in zoom-in duration-300">
          <div className="text-8xl mb-6">⚠️</div>
          <h2 className="text-4xl font-black mb-4 uppercase tracking-tighter">Stay Focused!</h2>
          <p className="text-xl mb-8 opacity-90">Leaving the exam screen is prohibited. Returning to the exam in...</p>
          <div className="text-9xl font-black tabular-nums">{warningTimer}</div>
          <p className="mt-8 text-sm font-bold uppercase tracking-widest bg-white/20 py-2 rounded-full">Automatic submission in progress</p>
        </div>
      </div>
    );
  }

  if (showConfirmation && submissionData) {
    return (
      <div className="max-w-lg mx-auto mt-20 text-center animate-in zoom-in duration-500">
        <Card className="bg-emerald-50 border-emerald-100 p-12">
          <div className="text-6xl mb-6">✅</div>
          <h2 className="text-3xl font-black text-emerald-900 mb-2">Exam Successfully Submitted</h2>
          <p className="text-emerald-700 mb-8">Thank you, {studentName}. Your answers have been recorded.</p>
          {submissionData.autoSubmitted && (
            <div className="bg-rose-100 text-rose-800 p-4 rounded-xl mb-8 font-bold text-sm">
              Note: This test was auto-submitted due to a tab-switch warning.
            </div>
          )}
          <Button onClick={() => onFinish(submissionData)} className="w-full py-4">View Final Grade</Button>
        </Card>
      </div>
    );
  }

  const currentQuestion = quiz.questions[currentIdx];

  return (
    <div className="max-w-3xl mx-auto pb-20 no-select">
      <div className="sticky top-4 z-10 flex justify-between items-center bg-white/95 backdrop-blur-sm p-4 rounded-2xl shadow-md border border-slate-100 mb-8">
        <div>
          <p className="text-[10px] font-black text-slate-400 uppercase tracking-widest">Active Assessment</p>
          <p className="font-bold text-slate-800 truncate max-w-[150px] md:max-w-xs">{quiz.title}</p>
        </div>
        <div className={`px-4 py-2 rounded-xl font-mono text-xl font-black shadow-inner transition-colors ${timeLeft < 60 ? 'bg-rose-100 text-rose-600 animate-pulse' : 'bg-slate-100 text-slate-700'}`}>
          {Math.floor(timeLeft / 60)}:{(timeLeft % 60).toString().padStart(2, '0')}
        </div>
      </div>

      <Card className="border-2 border-slate-50">
        <div className="mb-6 flex justify-between items-center">
          <span className="text-[10px] font-black text-rose-500 uppercase bg-rose-50 px-3 py-1 rounded-full">
            Question {currentIdx + 1} of {quiz.questions.length}
          </span>
          <span className="text-[10px] font-black text-slate-400 uppercase tracking-widest">
            ID: {quiz.id}
          </span>
        </div>
        
        <h3 className="text-2xl font-black text-slate-800 mb-8 leading-tight">
          {currentQuestion.question}
        </h3>

        <div className="space-y-3 mb-10">
          {currentQuestion.options.map((opt, i) => (
            <label 
              key={i} 
              className={`block p-5 rounded-2xl border-2 cursor-pointer transition-all duration-200 active:scale-[0.98] ${
                answers[currentQuestion.id] === opt 
                  ? 'border-rose-400 bg-rose-50 shadow-sm' 
                  : 'border-slate-100 hover:border-slate-200 bg-white'
              }`}
            >
              <input 
                type="radio" 
                className="hidden" 
                checked={answers[currentQuestion.id] === opt} 
                onChange={() => setAnswers({...answers, [currentQuestion.id]: opt})} 
              />
              <span className="flex items-center gap-4">
                <span className={`w-6 h-6 rounded-full border-2 flex items-center justify-center text-[10px] font-bold ${answers[currentQuestion.id] === opt ? 'bg-rose-500 border-rose-500 text-white' : 'border-slate-200 text-slate-400'}`}>
                  {String.fromCharCode(65 + i)}
                </span>
                <span className={`font-bold ${answers[currentQuestion.id] === opt ? 'text-rose-900' : 'text-slate-700'}`}>{opt}</span>
              </span>
            </label>
          ))}
        </div>

        <div className="flex justify-between items-center pt-6 border-t border-slate-50">
          <Button 
            variant="secondary" 
            onClick={() => setCurrentIdx(p => Math.max(0, p - 1))} 
            disabled={currentIdx === 0}
          >
            Previous
          </Button>
          
          <div className="flex gap-2">
            {currentIdx === quiz.questions.length - 1 ? (
              <Button variant="accent" onClick={() => {
                if (window.confirm("Are you sure you want to submit the exam?")) handleSubmit();
              }}>
                Finish & Submit
              </Button>
            ) : (
              <Button onClick={() => setCurrentIdx(p => p + 1)}>
                Next Question
              </Button>
            )}
          </div>
        </div>
      </Card>
      
      <p className="text-center mt-8 text-[10px] text-slate-400 font-bold uppercase tracking-[0.2em]">
        Do not close this window or switch tabs.
      </p>
    </div>
  );
};
